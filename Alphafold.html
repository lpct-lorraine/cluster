#!/bin/bash

#SBATCH --comment=test-alphafold
#SBATCH --job-name=test-alpha
#SBATCH --output=output.%j.alphafold-test
#SBATCH --error=output.%j.alphafold-test
#SBATCH -p lia1
#SBATCH --nodelist=cnl23,cnl05
#SBATCH -N 1
#SBATCH --ntasks=32
#SBATCH --gres=gpu:2

module load alphafold

export DOWNLOAD_DIR=/auto/lia/alphafold/alphafold_db
export TMPDIR=/scratch/

python3 /auto/soft/alphafold/alphafold/run_alphafold.py  \
--data_dir=$DOWNLOAD_DIR   \
--output_dir=/scratch/alphafold/test1  \
--fasta_paths=T1050.fasta   \
--max_template_date=2020-05-14  \
--use_gpu_relax='true'  \
--uniref90_database_path=/auto/lia/alphafold/alphafold_db/uniref90/uniref90.fasta  \
--mgnify_database_path=/auto/lia/alphafold/alphafold_db/mgnify/mgy_clusters_2018_12.fa  \

--obsolete_pdbs_path=/auto/lia/alphafold/alphafold_db/pdb_mmcif/obsolete.dat  \
--hhblits_binary_path=hhblits  \
--hhsearch_binary_path=hhsearch  \
