Slurm script

How to open an interactive session ? 

srun -p taras2 -n 12 -N1 --exclusive --exclude=cns82 --pty bash

No interactive session : 

#!/bin/bash
#SBATCH -p taras2
#SBATCH -n 12
#SBATCH -N 1
#SBATCH -t 10:00
#SBATCH -J test_NX
#SBATCH -o slurm.out
#SBATCH -e slurm.out
#SBATCH --exclusive
#SBATCH --exclude=cns82

set -x

module purge
module load  NewtonX/2.4-B06 gaussian/16.C01


ORIGINALDIR=${SLURM_SUBMIT_DIR}

WRK=/scratch/${USER}/tmp/${SLURM_JOB_ID}

mkdir -p $WRK
cd $WRK

$NX/inp-testnx.pl
