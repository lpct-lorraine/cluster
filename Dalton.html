Compilation of the latest version

Sequential version

[tmssoft@cns05 2018]$ export FC=ifort
[tmssoft@cns05 2018]$ export CC=icc
[tmssoft@cns05 2018]$ export CXX=icc

./setup
 
cd build

FC=ifort CC=icc CXX=icc cmake -DENABLE_MPI=OFF -DENABLE_SGI_MPT=OFF -DENABLE_OPENMP=OFF -DENABLE_64BIT_INTEGERS=OFF -DENABLE_GPU=OFF -DENABLE_CUBLAS=OFF -DENABLE_REAL_SP=OFF -DENABLE_STATIC_LINKING=OFF -DENABLE_SCALAPACK=OFF -DCMAKE_BUILD_TYPE=release /auto/soft/dalton/2018

make -j 10

Parallel version

module load compiler/intel19/u3
module load mpich/3.2-gnu-5.2.0/std
module load cmake/3.5.0-local-gnu-4.4.7/std

FC=mpifort CC=mpicc CXX=mpicc ./setup --mpi --omp
cd build/
make -j12

mkdir -p /scratch/tmssoft/tmp
export DALTON_TMPIDR=/scratch/tmssoft/tmp
export DALTON_LAUNCHER="mpirun -np 12"
export LSDALTON_LAUNCHER="mpirun -np 12"
make test
