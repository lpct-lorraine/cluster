SLURM script

Exemple de script "submit.sh" : 

#!/bin/bash
#SBATCH -p toul
#SBATCH -N 1
#SBATCH -n 1
#SBATCH -J test_tonto
#SBATCH -o slurm.out
#SBATCH -e slurm.out

module load tonto

SCRATCHDIR="/scratch/${USER}/tmp_${JOBID}/"
mkdir -p ${SCRATCHDIR} 

cp -r  ${TONTO}/tonto ${SCRATCHDIR}
cp -r  ${TONTO}/basis_sets ${SCRATCHDIR}


for file in stdin h2o.fchk; do 
  cp -av $file ${SCRATCHDIR};
done

cd ${SCRATCHDIR}

tonto


for file in stdout; do 
  cp -av  ${SCRATCHDIR}/$file ${SLURM_SUBMIT_DIR}/;
done


cd ${SLURM_SUBMIT_DIR}
