Built the python env

export INSTALLPATH=/auto/soft/TheoDORE/

cd $INSTALLPATH

virtualenv -p /usr/bin/python3 $INSTALLPATH/venv-python3

source $INSTALLPATH/venv_python3.6/bin/activate

pip install --upgrade pip

pip install numpy matplotlib

python-openbabel should be present in the OS.
