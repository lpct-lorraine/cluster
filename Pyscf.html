Running with a SLURM script

cat submit.sbatch 
#!/bin/bash
#SBATCH --ntasks=12
#SBATCH --job-name=pyscf-test
#SBATCH --output=log.%j.out
#SBATCH --error=log.%j.err
#SBATCH --time=02:00:00
#SBATCH -p bolo
#SBATCH -N 1

set -x

module purge
module load pyscf

source /auto/soft/pyscf/bin/activate

export OMP_NUM_THREADS=${SLURM_NTASKS}
python bz.py
python n2.py
